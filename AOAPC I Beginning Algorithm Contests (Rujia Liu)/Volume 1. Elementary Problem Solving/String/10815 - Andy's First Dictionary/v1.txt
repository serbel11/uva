#include <stdio.h>

int StrCmpWOCase(const char* s1, const char* s2){
	if (strlen(s1) != strlen(s2))
		return 1;
	else {
		int i;
		for (i = 0; i < strlen(s1); i++){
			if ((s1[i] - s2[i]) % 32 != 0)
				return 1;
		}
	}
	return 0;
}

void LettersToUpperCase(char* str){
	int j;
	for (j = 0; j < strlen(str); j++)
	{
		if ((str[j] >= 65) && (str[j] <= 90))
			str[j] = str[j] + 32;
	}
}
